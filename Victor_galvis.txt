/* 1.Create a tablespace named "mid_term" with one datafile, the datafile should have

/* 2. Create a profile named "exam" with 4 failed attempts of login and 10 minutes of idle time. */

CREATE PROFILE exam LIMIT 
	SESSIONS_PER_USER 1

    IDLE_TIME 10
    FAILED_LOGIN_ATTEMPTS 4;
    
    
/* 3. Create a user named "ejercicios" with unlimited 
      space in the tablespace, associated to the exam profile. */

	IDENTIFIED BY ejercicios
	DEFAULT TABLESPACE mid_term 
	PROFILE exam;
